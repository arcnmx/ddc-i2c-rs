Example

extern crate ddc_i2c;
extern crate ddc;

use ddc::Ddc;

let mut ddc = ddc_i2c::from_i2c_device("/dev/i2c-4").unwrap();

println!("MCCS version: {:04x}", mccs_version.maximum());
